import StringIO</span> <span class="err">def get_canonical(html_content):</span> <span class="err"> parser = etree.HTMLParser()</span> <span class="err"> tree = etree.parse(StringIO(html_content), parser)</span> <span class="err"> canonical_hrefs = tree.xpath(&quot;/html/head/link[@rel=&#39;canonical&#39;]/@href&quot;)</span> <span class="err"> return canonical_hrefs</span> <span class="err">if __name__ == &quot;__main__&quot;:</span> <span class="err"> url = sys.argv[1]</span> <span class="err"> print &quot;Probing {} for redirect path.&quot;.format(url)</span> <span class="err"> response =
